 326# Get project info
 327# ------------------------------------------------------------------------------
 328def get_project_info():
 329    """
 330    Get project info
 331
 332    Asks the user for project info, such as type and name, to be saved to
 333    G_DICT_SETTINGS.
 334    """
 335
 336    # ---------------------------------------------------------------
 337
 338    # first question is type
 339    # NB: keep asking until we get a response that is one of the types
 340
 341    # sanity check
 342    type_prj = ""
 343
 344    # get possible types
 345    list_types = [f"{key} ({val[0]})" for (key, val) in DICT_TYPES.items()]
 346    str_types = " | ".join(list_types)
 347    in_type = f"{G_STRINGS['S_PRJ_TYPE']} [{str_types}]: "
 348
 349    # loop forever until we get a valid type (or user presses Ctrl+C)
 350    while True:
 351        # ask for type of project
 352        type_prj = input(in_type)
 353
 354        # check for valid type
 355        if _check_type(type_prj):
 356            # at this point, type is valid so exit loop
 357            break
 358
 359    # save project type
 360    G_DICT_SETTINGS["__PP_TYPE_PRJ__"] = type_prj
 361
 362    # get output subdir
 363    dir_type = DICT_TYPES[type_prj][1]
 364
 365    # ---------------------------------------------------------------
 366
 367    # next question is name
 368    # NB: keep asking until we get a response that is a valid name and does not
 369    # already exist
 370
 371    # sanity check
 372    name_prj = ""
 373
 374    # loop forever until we get a valid name (or user presses Ctrl+C)
 375    while True:
 376        # ask for name of project
 377        name_prj = input(f"{G_STRINGS['S_PRJ_NAME']}: ")
 378
 379        # check for valid name
 380        if _check_name(name_prj):
 381            # set up for existence check
 382            tmp_dir = _DIR_BASE / dir_type / name_prj
 383
 384            # ------------------------------------------------------------------
 385
 386            # TODO: remove after debugging all project types
 387            if DEBUG:
 388                if tmp_dir.exists():
 389                    shutil.rmtree(tmp_dir)
 390
 391            # ------------------------------------------------------------------
 392
 393            # check if project already exists
 394            if tmp_dir.exists():
 395                print(G_STRINGS['S_NAME_ERR'].format(name_prj, dir_type))
 396            else:
 397                break
 398
 399    # calculate final project location
 400    global G_DIR_PRJ  # pylint: disable=global-statement
 401    # NB: weird thing about python and globals: you can use a global dict
 402    # WITHOUT using the 'global' keyword, as long as you're not assigning to the
 403    # actual dict variable, only its contents
 404    # in this case though, we are assigning to a scalar value, and so must use
 405    # the 'global' keyword above
 406    G_DIR_PRJ = _DIR_BASE / dir_type / name_prj
 407
 408    # at this point, both dir and name are valid, so store both and exit
 409    # NB: make sure to store the project dir as a string, so it can be saved
 410    # to file
 411    G_DICT_SETTINGS["__PP_DIR_PRJ__"] = str(G_DIR_PRJ)
 412    G_DICT_SETTINGS["__PP_NAME_BIG__"] = name_prj
 413
 414    # calculate small name
 415    info_name_small = name_prj.lower()
 416    G_DICT_SETTINGS["__PP_NAME_SMALL__"] = info_name_small
 417
 418    # calculate current date (format assumed from constants above)
 419    # NB: this gives us a create date that we cannot get reliably from Linux
 420    # later this may be changed to modified date
 421    now = datetime.now()
 422    fmt_date = DICT_USER["__PP_DATE_FMT__"]
 423    info_date = now.strftime(fmt_date)
 424    G_DICT_SETTINGS["__PP_DATE__"] = info_date

 494# ------------------------------------------------------------------------------
 495# Scan dirs/files in the project for replacing text
 496# ------------------------------------------------------------------------------
 497def do_fix():
 498    """
 499    Scan dirs/files in the project for replacing text
 500
 501    Scans for dirs/files under the project's location. For each dir/file it
 502    encounters, it passes the path to a filter to determine if the file
 503    needs fixing based on its appearance in the blacklist.
 504    """
 505
 506    # fix uo the blacklist and convert to absolute Path objects
 507    _fix_blacklist()
 508
 509    # just shorten the names
 510    skip_all = DICT_BLACKLIST["PP_SKIP_ALL"]
 511    skip_contents = DICT_BLACKLIST["PP_SKIP_CONTENTS"]
 512    skip_header = DICT_BLACKLIST["PP_SKIP_HEADER"]
 513    skip_text = DICT_BLACKLIST["PP_SKIP_TEXT"]
 514    skip_path = DICT_BLACKLIST["PP_SKIP_PATH"]
 515
 516    # walk from project dir
 517    # NB: topdown=False is required for the renaming, as we don't want to rename
 518    # (and thus clobber) a directory name before we rename all its child
 519    # dirs/files
 520    # it should have no effect on the other fixes
 521    for root, dirs, files in os.walk(G_DIR_PRJ, topdown=False):
 522        # NB: note that root is a full path, dirs and files are relative to root
 523
 524        # convert everything into Path objects
 525        root = Path(root)
 526        dirs = [root / d for d in dirs]
 527        files = [root / f for f in files]
 528
 529        # skip dir if in skip_all
 530        if root in skip_all:
 531            continue
 532
 533        # remove items is skip_all
 534        files = [item for item in files if item not in skip_all]
 535
 536        # for each file item (not dirs, they will be handled on the next iter)
 537        for item in files:
 538            # fix README if it is the top-level README.md
 539            # NB: need to do before any other stuff, requires special treatment
 540            if root == G_DIR_PRJ and item.name == README_NAME:
 541                _fix_readme(item)
 542
 543            # if we shouldn't skip contents
 544            if item not in skip_contents:
 545                # fix headers
 546                if item not in skip_header:
 547                    _fix_header(item)
 548
 549                # fix text
 550                if item not in skip_text:
 551                    _fix_text(item)
 552
 553            # fix path
 554            if item not in skip_path:
 555                _fix_path(item)
 556
 557        # fix current dir path
 558        if root not in skip_path:
 559            _fix_path(root)

 802# Replace dunders inside headers
 803# ------------------------------------------------------------------------------
 804def _fix_header(path):
 805    """
 806    Replace dunders inside headers
 807
 808    Arguments:
 809        path: Path for replacing header text
 810
 811    Replaces header text inside a file. Given a Path object, it iterates the
 812    lines one by one, replacing header text as it goes. When it is done, it
 813    saves the new lines as the old file. This replaces the __PP_.. stuff inside
 814    headers. Using this method, we preserve any right-aligned text in each
 815    header row (See the header in this file for an example). Adapting this
 816    function to your style of header should be easy using the regex pattern, and
 817    modifying DICT_HEADER to suit.
 818    NOTE: This method relies HEAVILY on the fact that the value exists, and that
 819    there is AT LEAST one space before and after the value, to work.
 820    Example (Dashes show required spaces):
 821    # --------------------------------------------------------------------------
 822    # Project:-__PP_NAME_BIG__-...
 823    # --------------------------------------------------------------------------
 824    This function will also preserve file extensions:
 825    # --------------------------------------------------------------------------
 826    # Filename:-__PP_NAME_SMALL__.py-...
 827    # --------------------------------------------------------------------------
 828    If your files do not use any right-aligned text, then the value can be empty
 829    and the space after the value is not necessary.
 830    """
 831
 832    # default lines
 833    lines = []
 834
 835    # open and read file
 836    with open(path, "r", encoding="UTF8") as a_file:
 837        lines = a_file.readlines()
 838
 839    # for each line in header
 840    for index, line in enumerate(lines):
 841        # for each header key
 842        for item in LIST_HEADER:
 843            # look for a header key
 844            pattern = (
 845                r"(\s*[#|<!--]\s*)"  # group 1: opening comment char(s)
 846                rf"({item})"  # group 2:key from LIST_HEADER
 847                r"f(\s*:\s+)"  # group 3:spaces, colon and spaces (at least 1)
 848                r"([^\s\.]*)"  # group 4: all not spaces and not dot (dunder)
 849                r"(\S*)"  # group 5: all not spaces (file ext)
 850                r"(\s*)"  # group 6: all spaces (padding)
 851                r"(.*)"  # group 7: all other chars (right-aligned text)
 852            )
 853
 854            # find first instance of header key
 855            res = re.search(pattern, line)
 856
 857            # no res, keep going
 858            if not res:
 859                continue
 860
 861            # value is group 4
 862            val = res.group(4)
 863
 864            # try to find key in dict_values
 865            # if no value or not in G_DICT_SETTINGS, leave it alone and keep
 866            # going
 867            if len(val) == 0 or val not in G_DICT_SETTINGS:
 868                continue
 869
 870            # get the value by backtracking into G_DICT_SETTINGS
 871            val = G_DICT_SETTINGS[val]
 872
 873            # count what we will need without spaces
 874            len_res = (
 875                len(res.group(1))
 876                + len(res.group(2))
 877                + len(res.group(3))
 878                + len(val)  # group 4 is val
 879                + len(res.group(5))
 880                + ""  # group 6 will be str_spaces
 881                + len(res.group(7))
 882            )
 883
 884            # make spaces
 885            # NB: this looks funky but it works:
 886            # 1. get the length of the old line (NOT always 80 since we can't
 887            # have trailing spaces)
 888            # 2. subtract 1 for newline
 889            # 3. subtract the other match lengths
 890            len_spaces = len(line) - 1 - len_res
 891            str_spaces = " " * len_spaces
 892
 893            # replace text in the line
 894            str_rep = rf"\g<1>\g<2>\g<3>{val}\g<5>{str_spaces}\g<7>"
 895            lines[index] = re.sub(pattern, str_rep, line)
 896
 897    # save lines to file
 898    with open(path, "w", encoding="UTF8") as a_file:
 899        a_file.writelines(lines)

1007    """
1008    Check project name for allowed characters
1009
1010    Arguments:
1011        name_prj: Name to check for allowed characters
1012
1013    Returns:
1014        Whether the name is valid to use
1015
1016    Checks the passed name for these criteria:
1017    1. non-blank name
1018    2. longer than 1 char
1019    3. starts with an alpha char
1020    4. ends with an alphanumeric char
1021    5. contains only alphanumeric chars or dash(-)/underscore (_)
1022    """
1023
1024    # NB: there is an easier way to do this with regex:
1025    # ^([a-zA-Z]+[a-zA-Z\d\-_]*[a-zA-Z\d]+)$ AND OMG DID IT TAKE A LONG
1026    # TIME TO FIND IT! in case you were looking for it. It will give you a quick
1027    # yes-no answer. I don't use it here because I want to give the user as much
1028    # feedback as possible, so I break down the regex into steps where each step
1029    # explains which part of the name is wrong.
1030
1031    # check for blank name
1032    if name_prj == "":
1033        return False
1034
1035    if len(name_prj) == 1:
1036        print(G_STRINGS['S_PRJ_LEN)'])
1037        return False
1038
1039    # match start or return false
1040    pattern = r"(^[a-zA-Z])"
1041    res = re.search(pattern, name_prj)
1042    if not res:
1043        print(G_STRINGS['S_PRJ_START'])
1044        return False
1045
1046    # match end or return false
1047    pattern = r"([a-zA-Z\d]$)"
1048    res = re.search(pattern, name_prj)
1049    if not res:
1050        print(G_STRINGS['S_PRJ_END'])
1051        return False
1052
1053    # match middle or return false
1054    pattern = r"(^[a-zA-Z\d\-_]*$)"
1055    res = re.search(pattern, name_prj)
1056    if not res:
1057        print(G_STRINGS['S_PRJ_CONTAIN'])
1058        return False
1059
1060    # if we made it this far, return true
1061    return True

def do_fix(): 
498def do_fix():
499    """
500    Scan dirs/files in the project for replacing text
501
502    Scans for dirs/files under the project's location. For each dir/file it
503    encounters, it passes the path to a filter to determine if the file
504    needs fixing based on its appearance in the blacklist.
505    """
506
507    # fix uo the blacklist and convert to absolute Path objects
508    _fix_blacklist()
509
510    # just shorten the names
511    skip_all = DICT_BLACKLIST["PP_SKIP_ALL"]
512    skip_contents = DICT_BLACKLIST["PP_SKIP_CONTENTS"]
513    skip_header = DICT_BLACKLIST["PP_SKIP_HEADER"]
514    skip_text = DICT_BLACKLIST["PP_SKIP_TEXT"]
515    skip_path = DICT_BLACKLIST["PP_SKIP_PATH"]
516
517    # walk from project dir
518    # NB: topdown=False is required for the renaming, as we don't want to rename
519    # (and thus clobber) a directory name before we rename all its child
520    # dirs/files
521    # it should have no effect on the other fixes
522    for root, dirs, files in os.walk(G_DIR_PRJ, topdown=False):
523        # NB: note that root is a full path, dirs and files are relative to root
524
525        # convert everything into Path objects
526        root = Path(root)
527        dirs = [root / d for d in dirs]
528        files = [root / f for f in files]
529
530        # skip dir if in skip_all
531        if root in skip_all:
532            continue
533
534        # remove items is skip_all
535        files = [item for item in files if item not in skip_all]
536
537        # for each file item (not dirs, they will be handled on the next iter)
538        for item in files:
539            # fix README if it is the top-level README.md
540            # NB: need to do before any other stuff, requires special treatment
541            if root == G_DIR_PRJ and item.name == README_NAME:
542                _fix_readme(item)
543
544            # if we shouldn't skip contents
545            if item not in skip_contents:
546                # fix headers
547                if item not in skip_header:
548                    _fix_header(item)
549
550                # fix text
551                if item not in skip_text:
552                    _fix_text(item)
553
554            # fix path
555            if item not in skip_path:
556                _fix_path(item)
557
558        # fix current dir path
559        if root not in skip_path:
560            _fix_path(root)
